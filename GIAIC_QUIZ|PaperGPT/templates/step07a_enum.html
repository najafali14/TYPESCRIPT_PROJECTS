TypeScript Quiz

1. What will be the value of c in the following TypeScript enum?


enum Color {Red, Green, Blue};

var c: Color = Color.Green;
console.log(c);

2. What will colorName contain in the following TypeScript enum?


enum Color1 {Red = 1, Green, Blue};
var colorName: string = Color1[2];
console.log(colorName);

3. What value will be assigned to colorIndex in the following TypeScript enum?


enum Color2 {Red = 1, Green = 2, Blue = 4};
var colorIndex = Color2["Blue"];
console.log(colorIndex);

4. What is the output of the following TypeScript code?


let x: { a: number, b: string };
let y: { a: number, b: string };
let z: { a: number, b: string };

x = { a: 1, b: "apple" };
y = { a: 2, b: "banana" };
z = { a: 3, b: "cherry" };

console.log(x.a + y.a + z.a);

5. What will be logged from the following TypeScript code?


let num: number = 5;

function multiply(value: number, multiplier: number): number {
    return value * multiplier;
}

console.log(multiply(num, 2));

6. What will be logged from the following TypeScript code?


let arr: number[] = [1, 2, 3, 4, 5];
let sum: number = arr.reduce((a, b) => a + b, 0);

console.log(sum);

7. What will be the type of myVar in the following TypeScript code?


let myVar: string | number;

myVar = "hello";

console.log(typeof myVar);

8. What will the following TypeScript code output?


let flag: boolean = true;

if (flag) {
    console.log("It's true!");
} else {
    console.log("It's false!");
}

9. What will be the output of the following TypeScript code?


let arr: string[] = ["apple", "banana", "cherry"];
let result: string = arr.join(", ");

console.log(result);

10. What will be the value of num after executing the following TypeScript code?


let num: number = 5;

function add(num1: number, num2: number): void {
    num1 += num2;
}

add(num, 3);

console.log(num);
